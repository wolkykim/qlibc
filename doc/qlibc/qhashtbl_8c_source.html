155 qhashtbl_t *qhashtbl(size_t range, int options) {
156  if (range == 0) {
157  range = DEFAULT_INDEX_RANGE;
158  }
159 
160  qhashtbl_t *tbl = (qhashtbl_t *) calloc(1, sizeof(qhashtbl_t));
161  if (tbl == NULL)
162  goto malloc_failure;
163 
164  // allocate table space
165  tbl->slots = (qhnobj_t **) calloc(range, sizeof(qhnobj_t *));
166  if (tbl->slots == NULL)
167  goto malloc_failure;
168 
169  // handle options.
170  if (options & QHASHTBL_THREADSAFE) {
171  Q_MUTEX_NEW(tbl->qmutex, true);
172  if (tbl->qmutex == NULL)
173  goto malloc_failure;
174  }
175 
176  // assign methods
177  tbl->put = put;
178  tbl->putstr = putstr;
179  tbl->putstrf = putstrf;
180  tbl->putint = putint;
181 
182  tbl->get = get;
183  tbl->getstr = getstr;
184  tbl->getint = getint;
185 
186  tbl->getnext = getnext;
187 
188  tbl->remove = remove_;
189 
190  tbl->size = size;
191  tbl->clear = clear;
192  tbl->debug = debug;
193 
194  tbl->lock = lock;
195  tbl->unlock = unlock;
196 
197  tbl->free = free_;
198 
199  // set table range.
200  tbl->range = range;
201 
202  return tbl;
203 
204  malloc_failure: errno = ENOMEM;
205  if (tbl) {
206  if (tbl->slots)
207  free(tbl->slots);
208  assert(tbl->qmutex == NULL);
209  free_(tbl);
210  }

227 static bool put(qhashtbl_t *tbl, const char *name, const void *data,
228  size_t size) {
229  if (name == NULL || data == NULL) {
230  errno = EINVAL;
231  return false;
232  }
233 
234  // get hash integer
235  uint32_t hash = qhashmurmur3_32(name, strlen(name));
236  int idx = hash % tbl->range;
237 
238  lock(tbl);
239 
240  // find existence key
241  qhnobj_t *obj;
242  for (obj = tbl->slots[idx]; obj != NULL; obj = obj->next) {
243  if (obj->hash == hash && !strcmp(obj->name, name)) {
244  break;
245  }
246  }
247 
248  // duplicate object
249  char *dupname = strdup(name);
250  void *dupdata = malloc(size);
251  if (dupname == NULL || dupdata == NULL) {
252  if (dupname != NULL)
253  free(dupname);
254  if (dupdata != NULL)
255  free(dupdata);
256  unlock(tbl);
257  errno = ENOMEM;
258  return false;
259  }
260  memcpy(dupdata, data, size);
261 
262  // put into table
263  if (obj == NULL) {
264  // insert
265  obj = (qhnobj_t *) calloc(1, sizeof(qhnobj_t));
266  if (obj == NULL) {
267  free(dupname);
268  free(dupdata);
269  unlock(tbl);
270  errno = ENOMEM;
271  return false;
272  }
273 
274  if (tbl->slots[idx] != NULL) {
275  // insert at the beginning
276  obj->next = tbl->slots[idx];
277  }
278  tbl->slots[idx] = obj;
279 
280  // increase counter
281  tbl->num++;
282  } else {
283  // replace
284  free(obj->name);
285  free(obj->data);
286  }
287 
288  // set data
289  obj->hash = hash;
290  obj->name = dupname;
291  obj->data = dupdata;
292  obj->size = size;
293 
294  unlock(tbl);
295  return true;
296 }

528 static bool getnext(qhashtbl_t *tbl, qhnobj_t *obj, const bool newmem) {
529  if (obj == NULL) {
530  errno = EINVAL;
531  return NULL;
532  }
533 
534  lock(tbl);
535 
536  bool found = false;
537 
538  qhnobj_t *cursor = NULL;
539  int idx = 0;
540  if (obj->name != NULL) {
541  idx = (obj->hash % tbl->range) + 1;
542  cursor = obj->next;
543  }
544 
545  if (cursor != NULL) {
546  // has link
547  found = true;
548  } else {
549  // search from next index
550  for (; idx < tbl->range; idx++) {
551  if (tbl->slots[idx] != NULL) {
552  cursor = tbl->slots[idx];
553  found = true;
554  break;
555  }
556  }
557  }
558 
559  if (cursor != NULL) {
560  if (newmem == true) {
561  obj->name = strdup(cursor->name);
562  obj->data = malloc(cursor->size);
563  if (obj->name == NULL || obj->data == NULL) {
564  DEBUG("getnext(): Unable to allocate memory.");
565  if (obj->name != NULL)
566  free(obj->name);
567  if (obj->data != NULL)
568  free(obj->data);
569  unlock(tbl);
570  errno = ENOMEM;
571  return false;
572  }
573  memcpy(obj->data, cursor->data, cursor->size);
574  obj->size = cursor->size;
575  } else {
576  obj->name = cursor->name;
577  obj->data = cursor->data;
578  }
579  obj->hash = cursor->hash;
580  obj->size = cursor->size;
581  obj->next = cursor->next;
582 
583  }
584 
585  unlock(tbl);
586 
587  if (found == false)
588  errno = ENOENT;
589 
590  return found;
591 }

604 static bool remove_(qhashtbl_t *tbl, const char *name) {
605  if (name == NULL) {
606  errno = EINVAL;
607  return false;
608  }
609 
610  lock(tbl);
611 
612  uint32_t hash = qhashmurmur3_32(name, strlen(name));
613  int idx = hash % tbl->range;
614 
615  // find key
616  bool found = false;
617  qhnobj_t *prev = NULL;
618  qhnobj_t *obj;
619  for (obj = tbl->slots[idx]; obj != NULL; obj = obj->next) {
620  if (obj->hash == hash && !strcmp(obj->name, name)) {
621  // adjust link
622  if (prev == NULL)
623  tbl->slots[idx] = obj->next;
624  else
625  prev->next = obj->next;
626 
627  // remove

629  free(obj->data);
630  free(obj);
631 
632  found = true;
633  tbl->num--;
634  break;
635  }
636 
637  prev = obj;
638  }
639 
640  unlock(tbl);
641 
642  if (found == false)
643  errno = ENOENT;
644 
645  return found;

664 void clear(qhashtbl_t *tbl) {
665  lock(tbl);
666  int idx;
667  for (idx = 0; idx < tbl->range && tbl->num > 0; idx++) {
668  if (tbl->slots[idx] == NULL)
669  continue;
670  qhnobj_t *obj = tbl->slots[idx];
671  tbl->slots[idx] = NULL;
672  while (obj != NULL) {
673  qhnobj_t *next = obj->next;
674  free(obj->name);
675  free(obj->data);
676  free(obj);
677  obj = next;
678 
679  tbl->num--;
680  }
681  }
682 
683  unlock(tbl);
684 }
